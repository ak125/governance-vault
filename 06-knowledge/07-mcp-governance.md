### PR Creation Logic

```typescript
// packages/mcp/src/governance/pr-creator.ts

export async function createPrFromArtifact(
  artifact: Artifact,
  rule: PrTriggerRule
): Promise<PullRequest> {
  const branchName = interpolateBranch(rule.branch, {
    date: new Date().toISOString().slice(0, 10),
    feature: artifact.name,
  });

  // 1. Create branch
  await git.checkout('-b', branchName);

  // 2. Write artifact to file
  const filePath = getArtifactPath(artifact);
  await fs.writeFile(filePath, artifact.content);

  // 3. Commit
  await git.add(filePath);
  await git.commit(`chore(auto): [${artifact.type}] ${artifact.name}

Generated by AI-COS skill.
Truth Level: ${artifact.metadata?.truthLevel || 'unknown'}

Co-Authored-By: Claude Opus 4.5 <noreply@anthropic.com>`);

  // 4. Push
  await git.push('origin', branchName);

  // 5. Create PR
  const pr = await github.pulls.create({
    owner: REPO_OWNER,
    repo: REPO_NAME,
    title: formatPrTitle(artifact),
    body: formatPrBody(artifact, rule),
    head: branchName,
    base: 'main',
    draft: rule.action === 'draft_pr',
  });

  // 6. Add labels
  await github.issues.addLabels({
    owner: REPO_OWNER,
    repo: REPO_NAME,
    issue_number: pr.number,
    labels: rule.labels,
  });

  // 7. Request reviewers
  await github.pulls.requestReviewers({
    owner: REPO_OWNER,
    repo: REPO_NAME,
    pull_number: pr.number,
    reviewers: rule.reviewers,
  });

  return pr;
}
```

## 2. CI Checks Pipeline

### GitHub Actions Workflow

```yaml
# .github/workflows/ai-artifact-check.yml

name: AI Artifact Validation

on:
  pull_request:
    labels:
      - auto-generated

jobs:
  detect-type:
    runs-on: ubuntu-latest
    outputs:
      type: ${{ steps.detect.outputs.type }}
    steps:
      - uses: actions/checkout@v4

      - name: Detect artifact type
        id: detect
        run: |
          if [[ -n $(git diff --name-only origin/main | grep '\.sql$') ]]; then
            echo "type=migration" >> $GITHUB_OUTPUT
          elif [[ -n $(git diff --name-only origin/main | grep 'seo-.*\.json$') ]]; then
            echo "type=seo" >> $GITHUB_OUTPUT
          elif [[ -n $(git diff --name-only origin/main | grep 'content-.*\.md$') ]]; then
            echo "type=content" >> $GITHUB_OUTPUT
          elif [[ -n $(git diff --name-only origin/main | grep '\.ts$') ]]; then
            echo "type=code" >> $GITHUB_OUTPUT
          fi

  migration-checks:
    needs: detect-type
    if: needs.detect-type.outputs.type == 'migration'
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - name: SQL Lint
        run: sqlfluff lint backend/supabase/migrations/

      - name: Migration Dry Run
        run: supabase db push --dry-run
        env:
          SUPABASE_ACCESS_TOKEN: ${{ secrets.SUPABASE_ACCESS_TOKEN }}

  seo-checks:
    needs: detect-type
    if: needs.detect-type.outputs.type == 'seo'
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - name: Setup Node
        uses: actions/setup-node@v4
        with:
          node-version: '20'

      - name: SEO Validation
        run: npm run seo:validate

      - name: URL Check (404s)
        run: npm run seo:check-urls

  content-checks:
    needs: detect-type
    if: needs.detect-type.outputs.type == 'content'
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - name: Spell Check
        run: npx cspell "**/*.md"

      - name: Truth Level Check
        run: |
          # Verify L4 content has review flag
          for f in $(find . -name "content-*.md"); do
            if grep -q "truthLevel: L4" "$f"; then
              if ! grep -q "reviewRequired: true" "$f"; then
                echo "ERROR: L4 content $f missing reviewRequired flag"
                exit 1
              fi
            fi
          done

  code-checks:
    needs: detect-type
    if: needs.detect-type.outputs.type == 'code'
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - name: Setup Node
        uses: actions/setup-node@v4
        with:
          node-version: '20'

      - name: Install Dependencies
        run: npm ci

      - name: Lint
        run: npm run lint

      - name: TypeCheck
        run: npm run typecheck

      - name: Test
        run: npm test

  auto-merge:
    needs: [migration-checks, seo-checks, content-checks, code-checks]
    if: |
      always() &&
      !contains(needs.*.result, 'failure') &&
      !contains(github.event.pull_request.labels.*.name, 'l4-review-required')
    runs-on: ubuntu-latest
    steps:
      - name: Add safe-to-merge label
        uses: actions/github-script@v7
        with:
          script: |
            await github.rest.issues.addLabels({
              owner: context.repo.owner,
              repo: context.repo.repo,
              issue_number: context.payload.pull_request.number,
              labels: ['safe-to-merge']
            })
```

## 3. Conventions

### Branch Naming

| Pattern | Usage |
|---------|-------|
| `auto/migrations/{YYYY-MM-DD}` | Database migrations |
| `auto/seo-fixes/{YYYY-MM-DD}` | SEO corrections |
| `auto/content/{slug}` | Generated content |
| `auto/code/{feature-name}` | Generated code |

### Commit Messages

```
chore(auto): [skill-name] description

- Brief summary of changes
- Impact on system

Generated by AI-COS.
Truth Level: L1/L2/L3/L4

Co-Authored-By: Claude Opus 4.5 <noreply@anthropic.com>
```

### Labels

| Label | Description |
|-------|-------------|
| `auto-generated` | All AI-generated PRs |
| `l4-review-required` | Content needing human validation |
| `safe-to-merge` | Passed all automated checks |
| `seo` | SEO-related changes |
| `database` | Database migrations |
| `code` | Code changes |
| `content` | Content changes |

### Auto-merge Conditions

Une PR peut être auto-mergée si:

1. ✅ Label `safe-to-merge` présent
2. ✅ Artifact avec Truth Level L1 ou L2
3. ✅ Tous les CI checks passent
4. ✅ Pas de label `l4-review-required`
5. ✅ Un seul type de changement (pas de mix migrations + content)

## 4. Truth Level Matrix

| Level | Auto-merge | Review | Example |
|-------|------------|--------|---------|
| L1 | ✅ Yes | Optional | Schema, generated code |
| L2 | ✅ Yes | Optional | Computed from DB |
| L3 | ❌ No | Spot-check | RAG-sourced content |
| L4 | ❌ No | Full review | AI-generated prose |
